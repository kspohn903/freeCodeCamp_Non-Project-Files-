function stoogeSort(arr) {
  let stoogeSortRecurse = (array, i, j) => {
    if (j == undefined) { j = array.length - 1; }
    if (i == undefined) { i = 0; }
    if (array[j] < array[i]) {
      [array[i], array[j]] = [array[j], array[i]];
    }
    if (j-i > 1) {
      let t = Math.floor((j-i+1)/3);
      stoogeSortRecurse(array,i,j-t);
      stoogeSortRecurse(array,i+t,j);
      stoogeSortRecurse(array,i,j-t);
  } };
  stoogeSortRecurse(arr);
  return arr;
}
