let leftFactorial = (n) => {
  if (n <= 1) { return n; }
  /* Note: for n>=20, the result may not be correct.
     This is bc JavaScript uses 53 bit integers and
     for n>=20 result becomes too large. */
  let res = 2, fact = 2;
  for (let i=2; i < n; i++) {
    res += fact;
    fact *= (i + 1);
  } return res; };
